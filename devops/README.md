
en pipeline groovy

```groovy
pipeline {
 agent any
 environment {
  SONARQUBE_URL = "http://172.17.0.1"
  SONARQUBE_PORT = "9000"
  SONAR_TOKEN = ""
 }
 options {
  skipDefaultCheckout()
 }
 tools {
    maven 'maven-3.9.4'
    node 'nodejs20'
  }
 stages {
  stage('SCM') {
   steps {
    checkout scm
   }
  }
  stage('Build') {
    steps {
    sh 'mvn clean package'
    }
  }
  stage('SonarQube') {
   steps {
    sh "mvn sonar:sonar -Dsonar.login=${SONAR_TOKEN} -Dsonar.host.url=$SONARQUBE_URL:$SONARQUBE_PORT -Dsonar.sourceEncoding=UTF-8 -Dsonar.sources=./ -Dsonar.sources=src"
   }
  }
 }
```
